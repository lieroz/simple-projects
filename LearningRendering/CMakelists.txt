cmake_minimum_required(VERSION 3.0)
project(LearningRendering)

set(CMAKE_CXX_STANDARD 20)

set(WIN_SOURCES
    window_win.cpp
    viddriver_d3d11.cpp
)

add_executable(main main.cpp)

if(WIN32)
    add_library(commonlib ${WIN_SOURCES})
    target_link_libraries(commonlib d3d11 d3dcompiler)
endif()

target_link_libraries(main PRIVATE commonlib)
